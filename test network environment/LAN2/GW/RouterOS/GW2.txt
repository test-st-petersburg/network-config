# устанавливаем имена интерфейсов для последующего использования

/interface ethernet
set 0 name="LAN2"
set 1 name="WAN"

# присваиваем ip адреса

/ip address
remove [find]
add interface="LAN2" address="192.168.2.1/24"
add interface="WAN" address="10.0.2.1/24"

# добавляем статические маршруты

# шлюз по умолчанию, в реальности не потребуется, должны получить с DHCP провайдера
/ip route
remove [find static=yes]
add dst-address=0.0.0.0/0 gateway=10.0.2.100 distance=100

# включаем мониторинг статуса интерфейсов

/interface detect-internet
set detect-interface-list=all

state print

# поднимаем DHCP сервер и настраиваем его

/ip pool
remove [find]
add name=LAN2 ranges=192.168.2.2-192.168.2.254
/ip dhcp-server option sets
remove [find]
add name=LAN2 options=""
/ip dhcp-server network
remove [find]
/ip dhcp-server network
remove [find]
add address=192.168.2.0/24 gateway=192.168.2.1 dns-none=yes dhcp-option-set=LAN2
/ip dhcp-server
remove [find]
add name=LAN2 interface=LAN2 address-pool=LAN2 authoritative=yes lease-time=1d
enable LAN2

# поднимаем NAT

/ip firewall nat
remove [find]
add action=masquerade chain=srcnat out-interface=WAN

# настраиваем firewall

/ip firewall filter
remove [find]
#разрешает входящий трафик от уже установленных соединений
add chain=input connection-state=established,related action=accept comment="enable incoming traffic from established connections"
add chain=forward connection-state=established,related action=accept
add chain=input connection-state=new protocol=icmp action=accept comment="enable incoming ICMP"
add chain=output connection-state=new protocol=icmp action=accept comment="enable outgoing ICMP"
add chain=input connection-state=new in-interface=LAN2 action=accept comment="enable LAN traffic"
add chain=forward connection-state=new in-interface=LAN2 action=accept comment="enable from LAN to WAN"

add chain=input action=drop

# настраиваем IPsec туннель и файрвол под него

/ip ipsec profile
remove [find default=no and dynamic=no]
add name=LAN1 dh-group=modp1024 enc-algorithm=3des
#nat-traversal=yes

/ip ipsec proposal
remove [find default=no and dynamic=no]
add name=LAN1 enc-algorithms=3des pfs-group=modp1024 auth-algorithms=sha1 disabled=no

/ip ipsec peer
remove [find]
add name=LAN1 address=10.0.1.1/32 profile=LAN1 disabledf=yes

/ip ipsec identity
remove [find dynamic=no]
add peer=LAN1 secret=535353 disabled=no

/ip ipsec policy
remove [find default=no and dynamic=no]
add src-address=192.168.2.0/24 src-port=any dst-address=192.168.1.0/24 dst-port=any sa-src-address=10.0.2.1 sa-dst-address=10.0.1.1 tunnel=yes action=encrypt proposal=LAN1 level=require ipsec-protocols=esp

/ip firewall nat
	add chain=srcnat action=accept \
		src-address=192.168.2.0/24 dst-address=192.168.1.0/24 \
		place-before=[find chain=srcnat and action=masquerade]

/ip route
	add dst-address=192.168.1.0/24 \
		pref-src=192.168.2.1 gateway=WAN distance=1

/ip ipsec peer enable LAN1

/ip ipsec
remote-peers print
installed-sa print
